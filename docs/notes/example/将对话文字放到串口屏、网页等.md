## Nodejs 代码
 
```js
const espAi = require("esp-ai"); 
const config = { 
    // 将语音识别的文字发回客户端
    onIATcb({ sendToClient }) {
        sendToClient();
    }, 
    // 将LLM推理的文字发回客户端
    onLLMcb({ sendToClient }) {
        sendToClient();
    },
};

const espAiIns = espAi(config);
```
